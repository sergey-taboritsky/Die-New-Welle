	on_declare_war = {
		effect = {
			if = {
				limit = {
					ROOT = {
						tag = FRA
					}
					FROM = {
						tag = GER
					}
				}
				set_global_flag = the_third_PF_war
				every_possible_country = {
					news_event = {
						id = AUS_decision_news_events.2
						hours = 12
						days = 1
					}
				}
			}
		}
